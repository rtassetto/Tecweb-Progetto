Account{
	*username varchar(20)
	password varchar(20)
	email  unique varchar(40)
	admin bool
	datacreazione datetime
}

PurchaseHistory{
	*orderid int
	-username(Account)
	-id(Prodotto)
	data datetime
	quantita int
}

Carrello{
	*(-username(Account)
	-id(Prodotto))
	quantita int
}

Recensione{
	*(-username(Account)
	-id(Prodotto))
	review varchar(500/1000)
	voto int
	data datetime
}

Prodotto{
	*id
	nome varchar(40)
	categoria enum
	desc varchar(500)
	Valutazione int
	prezzo float / int int
}

Bundles{

	desc varchar(300)
}

Bundleparts{
